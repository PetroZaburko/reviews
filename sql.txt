CREATE TABLE reviews (
	id int auto_increment primary key,
	title varchar(191) not null,
	review text not null,
	nickname varchar(191) not null,
	email varchar(191) not null,
	pros varchar(255),
	cons varchar(255),
	`date` varchar(191),
	recommended tinyint
);

CREATE TABLE photos (
id int auto_increment primary key,
photo varchar(191),
review_id int
);
